import numpy as np
from matplotlib import pyplot as plt

def plot_hr_diagram(magnitudes, colors):
    """A sample function to plot a color-magnitude diagram."""
    plt.figure(figsize=(8, 10))
    plt.scatter(colors, magnitudes, s=1, alpha=0.5, c='k')
    plt.ylim(max(magnitudes), min(magnitudes)) # Invert y-axis
    plt.xlabel('Color (G_BP - G_RP)')
    plt.ylabel('Absolute Magnitude (M_G)')
    plt.title('Hertzsprung-Russell Diagram')
    plt.grid(True, alpha=0.2)
    plt.show()
